155{

159 // コマンドリストをリセット
160 commandList.Get()->Reset(commandAllocator.Get(), pipeline.Get());
161
162 //ビューポートのセット
163 commandList.Get()->RSSetViewports(1, &viewport);
164 //シザーのセット
165 commandList.Get()->RSSetScissorRects(1, &scissorRect);
166
167 BufferManager::Instance()->ResetUseCounter();
168
169 basicHeap.ResetCounter();
170
171 // バックバッファのインデックスを格納
172 frameIndex = swapChain.Get()->GetCurrentBackBufferIndex();
173
174 // バックバッファをレンダーターゲットとして使用
175 auto barrier = CD3DX12_RESOURCE_BARRIER::Transition(renderTargets[frameIndex].Get(), D3D12_RESOURCE_STATE_PRESENT, D3D12_RESOURCE_STATE_RENDER_TARGET);
176 commandList.Get()->ResourceBarrier(1, &barrier);
177 // リソースバリアとは、GPU側で扱うリソースの状況を同期させる機能。
178 // マルチスレッドを前提とした動きなので、GPU側の動作も複数のアクセスが同時に行われることを想定した機能だということ。

180 // レンダーターゲットビューのハンドルを作成
181 CD3DX12_CPU_DESCRIPTOR_HANDLE rtvHandle(rtvHeap->GetCPUDescriptorHandleForHeapStart(), frameIndex, rtvDescriptorSize);
182 commandList.Get()->OMSetRenderTargets(1, &rtvHandle, false, nullptr);
183
184 //バックバッファに描画(コマンドを記録)
185 const FLOAT clearColor[] = { 0.0f, 0.2f, 0.4f, 1.0f }; // 青っぽい色
186 commandList.Get()->ClearRenderTargetView(rtvHandle, clearColor, 0, nullptr);
187
188 if (pipeline.Get() == nullptr)
189 {
190 return;
191 }
192
193 commandList.Get()->SetGraphicsRootSignature(rootSignature.Get());
194 commandList.Get()->SetPipelineState(pipeline.Get());

394 matrix.r[1].m128_f32[1] = -2.0f / viewport.Height;

397 matrix.r[3].m128_f32[1] = 1.0f;

400 constBuff->Map(0, nullptr, (void**)&mapMatrix); //マップ
